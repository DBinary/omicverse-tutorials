# Sphinx documentation dependencies for omicverse-tutorials
#
# Pin sphinx to the 7.x line. The 8.x line threw an
# "issubclass() arg 2 must be a class, a tuple of classes, or a union"
# error at autosummary.process_generate_options on python 3.11 — the pin
# above 7 keeps us on the classic autosummary code path that surfaces the
# real error instead of swallowing it.
sphinx>=7.4,<8.0
sphinx-book-theme>=1.1,<1.2
myst-nb>=1.1,<1.2
sphinx-autodoc-typehints<2.4
sphinx-copybutton
sphinx-design<0.7
sphinxext-opengraph
sphinx-hoverxref<1.5
linkify-it-py

ipykernel
nbformat

# Scientific stack (for autodoc imports).
#
# scipy 1.16 added the experimental `scipy.stats._new_distributions` module
# whose import side-effect calls `_combine_docs` → `_generate_example` →
# `xp_promote` → `is_torch_array(arg)` → `_issubclass_fast(cls, "torch",
# "Tensor")`, which raises `TypeError: issubclass() arg 2 must be a class`
# when torch isn't installed. That fires whenever sklearn's import chain
# pulls in scipy.stats, which any autodoc of `omicverse.bulk` triggers.
# Pin scipy below 1.16 to dodge the experimental module entirely.
numpy
pandas
matplotlib
anndata
scanpy
scipy>=1.11,<1.16
liana
